def void main ():
	int n, result
	int preOldResult, oldResult
	int i
	
	n = 5

	if(n <= 0):
		result = 0

	if(n > 0 && n < 3):
		result = 1

	result = 0;
	preOldResult = 1;
	oldResult = 1;

	i = 2
	while (i < 2):
		result = preOldResult + oldResult
		preOldResult = oldResult
		oldResult = result
		i = i + 1

	print(result)
